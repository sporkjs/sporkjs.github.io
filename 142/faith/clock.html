Clock

Faith Hoppen

Math 142, Fall 2013

Assignment: In this project, we were asked to make an analog clock that would tell the correct time and run when the page was opened.


Tools: In order to complete this, we used the variable, function, loop, hour, minute, and second functions from JavaScript programming and the Spork editor to make it come to life.

Code:

//Faith Hoppen
//Math 142, Fall 2013
//Analog Clock


function drawTicker(l,w){
  color(0,153,153);
  linewidth(l);                           
  line(0,0,w-2,0);
  color(0,153,153);
  linewidth(1);
  line(w-15,0,w-15,15,w,0,w-15,-15);
  fillcolor(0,153,153);
  fill();
}

function secondTicker(r,s){
  color(0,153,153);
  linewidth(r);
  line(0,0,s-2,0);
}

function drawSecond(){
  save();
  var x=second();
  linewidth(1);
  rotate(x/60*360-90);
  secondTicker(1,190);
  restore();
}

function drawMinute(){
  save();
  var y=minute();
  linewidth(2);
  rotate(y/60*360-90);
  drawTicker(2,170);
  restore();
}

function drawHour(){
  save();
  var z=hour()%12;
  linewidth(3);
  rotate(z/12*360-90);
  drawTicker(3,100);
  restore();
}

function drawClock(){
  save();
  color(0,153,153);
  linewidth(1.5);
  circle(0,0,200); 
  stroketext("12", -23,-165);
  stroketext("3",165,13);
  stroketext("6",-13,193);
  stroketext("9",-193,13);
  restore();
}

function setup(){
  translate(200,200);
  timestep(500);
  loop();
}

function draw(){
  clear();
  rect(-200,-200,400,400);
  fillcolor(153,255,153);
  fill();
  drawClock();
  drawSecond();
  drawMinute();
 drawHour();
}
